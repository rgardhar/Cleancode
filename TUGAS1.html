PERHITUNGAN BALOK

            
                /**
                * Kelas untuk menghitung volume dan luas permukaan balok.
                */
                public class Balok {
           
                    private double panjang;
                    private double lebar;
                    private double tinggi;
                
                    /**
                     * @param panjang Panjang dari balok
                     * @param lebar   Lebar dari balok
                     * @param tinggi  Tinggi dari balok
                     */
                    public Balok(double panjang, double lebar, double tinggi) {
                        this.panjang = panjang;
                        this.lebar = lebar;
                        this.tinggi = tinggi;
                    }
                
                    /**
                     * Menghitung volume dari balok.
                     * Volume balok dihitung dengan rumus: panjang * lebar * tinggi
                     * @return Volume balok dalam satuan yang sama dengan panjang, lebar, dan tinggi.
                     */
                    public double hitungVolume() {
                        return panjang * lebar * tinggi;
                    }
     
                    /**
                     * Menghitung luas permukaan dari balok.
                     * Luas permukaan balok dihitung dengan rumus: 2 * (panjang * lebar + panjang * tinggi + lebar * tinggi) 
                     * @return Luas permukaan balok dalam satuan yang sama dengan panjang, lebar, dan tinggi.
                     */
                     public double hitungLuasPermukaan() {
                         return 2 * (panjang * lebar + panjang * tinggi + lebar * tinggi);
                     }
     
                    /**
                     * Menampilkan informasi mengenai volume, dan luas permukaan balok.
                     */
                    public void tampilkanInformasi() {
                        System.out.println("Volume: " + hitungVolume() + " satuan^3");
                        System.out.println("Luas Permukaan: " + hitungLuasPermukaan() + " satuan^2");
                    }
                }
